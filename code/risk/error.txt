import pandas as pd
import matplotlib.pyplot as plt

# Load the data
risk_df = pd.read_csv('risk_level.csv')

# Assuming the risk level values are already scaled to 1 (low), 2 (moderate), 3 (high)
risk_counts = risk_df['Risk Level'].value_counts().sort_index()

# Print the counts
print("Risk Level Counts:")
print(risk_counts)

# Graph the categorization
plt.bar(risk_counts.index, risk_counts.values)
plt.xlabel('Risk Level')
plt.ylabel('Number of Providers')
plt.title('Provider Risk Level Distribution')
plt.xticks([1, 2, 3], ['Low (1)', 'Moderate (2)', 'High (3)'])
plt.show()
